## Главная идея

Модуль `os` даёт доступ к операционной системе: пути, папки, переменные среды, информация о системе. Это мост между Python и Windows.

────────────────────
## Импорт и базовые операции

```python
import os

print(os.getcwd())              # текущая папка
print(os.name)                  # → nt  (Windows)
print(os.sep)                   # → \   (разделитель путей Windows)
```

## Работа с путями — os.path

```python
import os

# Сборка пути (правильно для любой ОС)
path = os.path.join("C:\\Users", "User", "AppData", "Temp")
print(path)   # → C:\Users\User\AppData\Temp

# Проверки
print(os.path.exists(path))    # True если путь существует
print(os.path.isfile(path))    # True если это файл
print(os.path.isdir(path))     # True если это папка

# Разбор пути
full = "C:\\Users\\User\\report.txt"
print(os.path.dirname(full))   # → C:\Users\User
print(os.path.basename(full))  # → report.txt
print(os.path.splitext(full))  # → ('C:\\Users\\User\\report', '.txt')
print(os.path.getsize(full))   # размер в байтах
```

## Работа с папками

```python
# Создать папку
os.makedirs("logs/2026", exist_ok=True)   # exist_ok=True — не ругаться если уже есть

# Список содержимого
for item in os.listdir("C:\\"):
    print(item)

# Обход дерева папок
for root, dirs, files in os.walk("C:\\Users\\User\\Documents"):
    for file in files:
        full_path = os.path.join(root, file)
        print(full_path)

# Удалить файл / папку
os.remove("old_log.txt")        # удалить файл
os.rmdir("empty_folder")        # удалить пустую папку
```

## Практика — PC Booster

```python
import os
import json

def get_system_paths() -> dict:
    """Возвращает важные системные пути Windows."""
    return {
        "temp":     os.environ.get("TEMP", ""),
        "appdata":  os.environ.get("APPDATA", ""),
        "windir":   os.environ.get("WINDIR", "C:\\Windows"),
        "username": os.environ.get("USERNAME", ""),
        "userprofile": os.environ.get("USERPROFILE", ""),
    }

def scan_temp_folder() -> dict:
    """Сканирует папку Temp и считает размер."""
    temp = os.environ.get("TEMP", "")
    if not os.path.exists(temp):
        return {"error": "Папка не найдена"}

    total_size  = 0
    file_count  = 0
    errors      = 0

    for root, dirs, files in os.walk(temp):
        for fname in files:
            fpath = os.path.join(root, fname)
            try:
                total_size += os.path.getsize(fpath)
                file_count += 1
            except (PermissionError, FileNotFoundError):
                errors += 1

    return {
        "path":       temp,
        "files":      file_count,
        "size_mb":    round(total_size / 1024 / 1024, 1),
        "errors":     errors,
    }

def setup_app_folder(app_name: str) -> str:
    """Создаёт папку приложения в AppData."""
    appdata = os.environ.get("APPDATA", "")
    app_dir = os.path.join(appdata, app_name)
    os.makedirs(app_dir, exist_ok=True)
    return app_dir

# Использование
paths = get_system_paths()
print(f"Пользователь: {paths['username']}")
print(f"Temp: {paths['temp']}")

app_folder = setup_app_folder("PCBooster")
print(f"Папка приложения: {app_folder}")

result = scan_temp_folder()
print(f"\nСканирование Temp:")
print(f"  Файлов:  {result['files']}")
print(f"  Размер:  {result['size_mb']} MB")
```

## Задание

PC Booster сканирует папку загрузок для очистки. Напиши функцию `scan_downloads()`.

**Дано:**
- Путь к папке загрузок: `os.path.join(os.environ["USERPROFILE"], "Downloads")`

**Нужно:**
1. Обойти все файлы в папке через `os.walk()`
2. Посчитать файлы по расширениям (`.exe`, `.zip`, `.pdf`, прочие)
3. Найти самый большой файл
4. Вывести отчёт

**Ожидаемый вывод:**
```
=== Папка загрузок ===
Всего файлов: 47
  .exe   — 12 файлов
  .zip   —  8 файлов
  .pdf   —  5 файлов
  прочие — 22 файла
Самый большой: setup_photoshop.exe (1847.3 MB)
```

💡 Подсказка 1
> Используй `os.path.splitext(filename)[1].lower()` чтобы получить расширение файла.

💡 Подсказка 2
> Для подсчёта по расширениям — словарь: `ext_count[ext] = ext_count.get(ext, 0) + 1`

💡 Подсказка 3
> Самый большой файл — накапливай `max_size` и `max_name` в цикле.

## Решение

```python
import os

def scan_downloads() -> dict:
    """Сканирует папку загрузок и считает статистику."""
    downloads = os.path.join(os.environ.get("USERPROFILE", ""), "Downloads")

    if not os.path.exists(downloads):
        return {"error": "Папка загрузок не найдена"}

    ext_count = {}
    total_files = 0
    max_size = 0
    max_name = ""

    for root, dirs, files in os.walk(downloads):
        for fname in files:
            full_path = os.path.join(root, fname)
            ext = os.path.splitext(fname)[1].lower() or "без расширения"
            ext_count[ext] = ext_count.get(ext, 0) + 1
            total_files += 1

            try:
                size = os.path.getsize(full_path)
                if size > max_size:
                    max_size = size
                    max_name = fname
            except (PermissionError, FileNotFoundError):
                continue

    return {
        "total": total_files,
        "extensions": ext_count,
        "max_name": max_name,
        "max_mb": round(max_size / 1024 / 1024, 1),
    }

# Использование
result = scan_downloads()

if "error" in result:
    print(result["error"])
else:
    print("=== Папка загрузок ===")
    print(f"Всего файлов: {result['total']}")
    for ext, count in sorted(result["extensions"].items(), key=lambda x: -x[1]):
        label = ext if ext != "без расширения" else "прочие"
        print(f"  {label:<10} — {count:>3} файлов")
    print(f"Самый большой: {result['max_name']} ({result['max_mb']} MB)")
```

**Разбор:**
1. `os.path.splitext(fname)[1]` → разбивает имя на `(имя, расширение)`, берём `[1]` → `.exe`
2. `ext_count.get(ext, 0) + 1` → безопасный счётчик: если ключа нет, возвращает 0
3. `os.path.getsize(full_path)` → размер файла в байтах, делим на `1024**2` для MB
4. `try/except PermissionError` → некоторые файлы защищены, пропускаем их
5. `sorted(..., key=lambda x: -x[1])` → сортировка по количеству (убывание)

## Заметки на полях

💡 os.path.join — всегда вместо конкатенации
> ```python
> # Плохо — не работает на разных ОС
> path = "C:\\Users\\" + username + "\\Desktop"
>
> # Хорошо
> path = os.path.join("C:\\Users", username, "Desktop")
> ```

⚠️ PermissionError
> Некоторые системные папки защищены. Всегда оборачивай `os.walk()` в `try/except PermissionError`.

📝 pathlib — современная альтернатива os.path
> ```python
> from pathlib import Path
> p = Path("C:/Users") / "User" / "Desktop"
> print(p.exists())
> print(p.suffix)    # расширение
> ```
> `pathlib` — более современный API, встречается в новом коде.
