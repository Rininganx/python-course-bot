## Главная идея

Планировщик задач Windows позволяет запускать скрипты по расписанию: при входе в систему, каждый час, в определённое время. Через `schtasks` и Python можно создавать, удалять и просматривать задачи.

────────────────────
## Просмотр задач через schtasks

```python
import subprocess

def run_schtasks(args: list) -> str:
    result = subprocess.run(
        ["schtasks"] + args,
        capture_output=True, text=True, encoding="cp866"
    )
    return result.stdout

# Список всех задач
print(run_schtasks(["/Query", "/FO", "LIST"]))

# Список в CSV формате
print(run_schtasks(["/Query", "/FO", "CSV", "/NH"]))

# Информация о конкретной задаче
print(run_schtasks(["/Query", "/TN", "\\Microsoft\\Windows\\Defrag\\ScheduledDefrag"]))
```

## Создание задачи

```python
import subprocess
import os

def create_task(
    task_name: str,
    script_path: str,
    trigger: str = "ONLOGON",
    run_level: str = "HIGHEST"
) -> bool:
    """
    Создаёт задачу в планировщике.
    trigger: ONLOGON | DAILY | HOURLY | ONSTART
    """
    cmd = [
        "schtasks", "/Create",
        "/TN", task_name,
        "/TR", f'python "{script_path}"',
        "/SC", trigger,
        "/RL", run_level,
        "/F",   # перезаписать если существует
    ]

    result = subprocess.run(
        cmd, capture_output=True, text=True, encoding="cp866"
    )
    return result.returncode == 0

def delete_task(task_name: str) -> bool:
    """Удаляет задачу из планировщика."""
    result = subprocess.run(
        ["schtasks", "/Delete", "/TN", task_name, "/F"],
        capture_output=True, text=True, encoding="cp866"
    )
    return result.returncode == 0

def run_task_now(task_name: str) -> bool:
    """Запускает задачу немедленно."""
    result = subprocess.run(
        ["schtasks", "/Run", "/TN", task_name],
        capture_output=True, text=True, encoding="cp866"
    )
    return result.returncode == 0
```

## Расписания (триггеры)

Триггер • Ключ • Описание
При входе • `ONLOGON` • При каждом входе пользователя
При старте • `ONSTART` • При загрузке Windows
Ежедневно • `DAILY` • Каждый день в заданное время
Каждый час • `HOURLY` • Каждый час
Раз в неделю • `WEEKLY` • По дням недели
Раз в месяц • `MONTHLY` • По числу месяца

## Автозапуск PC Booster

```python
import subprocess
import sys
import os

APP_NAME    = "PCBooster"
TASK_NAME   = "\\PCBooster\\AutoOptimize"

def install_autostart() -> bool:
    """Добавляет PC Booster в автозапуск через планировщик."""
    script = os.path.abspath(sys.argv[0])

    result = subprocess.run([
        "schtasks", "/Create",
        "/TN", TASK_NAME,
        "/TR", f'"{sys.executable}" "{script}" --silent',
        "/SC", "ONLOGON",
        "/RL", "HIGHEST",
        "/F",
    ], capture_output=True, text=True, encoding="cp866")

    if result.returncode == 0:
        print(f"Автозапуск установлен: {TASK_NAME}")
        return True
    else:
        print(f"Ошибка: {result.stderr}")
        return False

def remove_autostart() -> bool:
    """Удаляет PC Booster из автозапуска."""
    result = subprocess.run(
        ["schtasks", "/Delete", "/TN", TASK_NAME, "/F"],
        capture_output=True, text=True, encoding="cp866"
    )
    success = result.returncode == 0
    print("Автозапуск удалён" if success else "Задача не найдена")
    return success

def check_autostart() -> bool:
    """Проверяет установлен ли автозапуск."""
    result = subprocess.run(
        ["schtasks", "/Query", "/TN", TASK_NAME],
        capture_output=True, text=True, encoding="cp866"
    )
    return result.returncode == 0

print(f"Автозапуск активен: {check_autostart()}")
```

## Практика — Планировщик очисток

```python
import subprocess

def schedule_cleanup(time_str: str = "03:00") -> bool:
    """
    Планирует ежедневную очистку temp в 3 ночи.
    time_str: время в формате HH:MM
    """
    cleanup_script = os.path.join(
        os.environ.get("APPDATA", ""), "PCBooster", "cleanup.py"
    )

    result = subprocess.run([
        "schtasks", "/Create",
        "/TN", "\\PCBooster\\DailyCleanup",
        "/TR", f'"{sys.executable}" "{cleanup_script}"',
        "/SC", "DAILY",
        "/ST", time_str,
        "/RL", "HIGHEST",
        "/F",
    ], capture_output=True, text=True, encoding="cp866")

    return result.returncode == 0
```

## Задание

PC Booster управляет расписанием через планировщик задач. Напиши текстовый интерфейс.

**Дано:**
- Задача в планировщике называется `PCBooster_AutoClean`

**Нужно:**
1. Текстовое меню с 5 пунктами
2. Каждый пункт вызывает свою функцию
3. Выводит результат (успех/ошибка)
4. Повторяет меню пока не выбран "Выход"

**Ожидаемый вывод:**
```
=== Планировщик PC Booster ===
1. Показать задачи
2. Включить автозапуск
3. Отключить автозапуск
4. Запустить очистку сейчас
0. Выход

Выбор: 1
Задача не найдена — автозапуск не настроен

Выбор: 0
Выход
```

💡 Подсказка 1
> Для проверки задачи: `schtasks /Query /TN PCBooster_AutoClean` — если задачи нет, `returncode != 0`.

💡 Подсказка 2
> Для создания: `schtasks /Create /TN ... /TR "python main.py" /SC ONLOGON /F` — `/F` перезаписывает.

💡 Подсказка 3
> Для удаления: `schtasks /Delete /TN PCBooster_AutoClean /F` — `/F` без подтверждения.

## Решение

```python
import subprocess

TASK_NAME = "PCBooster_AutoClean"
SCRIPT_PATH = "C:\\PCBooster\\main.py"

def run_schtasks(args: list) -> tuple:
    """Запускает schtasks и возвращает (returncode, stdout)."""
    result = subprocess.run(
        ["schtasks"] + args,
        capture_output=True, text=True, encoding="cp866"
    )
    return result.returncode, result.stdout

def show_tasks():
    """Показывает статус задачи."""
    code, output = run_schtasks(["/Query", "/TN", TASK_NAME])
    if code == 0:
        print(f"Задача найдена:\n{output.strip()}")
    else:
        print("Задача не найдена — автозапуск не настроен")

def enable_autostart():
    """Создаёт задачу автозапуска."""
    code, _ = run_schtasks([
        "/Create", "/TN", TASK_NAME,
        "/TR", f'python "{SCRIPT_PATH}"',
        "/SC", "ONLOGON", "/F"
    ])
    print("Автозапуск включён!" if code == 0 else "Ошибка создания задачи")

def disable_autostart():
    """Удаляет задачу автозапуска."""
    code, _ = run_schtasks(["/Delete", "/TN", TASK_NAME, "/F"])
    print("Автозапуск отключён!" if code == 0 else "Задача не найдена")

def run_now():
    """Запускает очистку немедленно."""
    code, _ = run_schtasks(["/Run", "/TN", TASK_NAME])
    print("Очистка запущена!" if code == 0 else "Задача не найдена — нечего запускать")

while True:
    print("\n=== Планировщик PC Booster ===")
    print("1. Показать задачи")
    print("2. Включить автозапуск")
    print("3. Отключить автозапуск")
    print("4. Запустить очистку сейчас")
    print("0. Выход")

    choice = input("\nВыбор: ").strip()

    if choice == "1":
        show_tasks()
    elif choice == "2":
        enable_autostart()
    elif choice == "3":
        disable_autostart()
    elif choice == "4":
        run_now()
    elif choice == "0":
        print("Выход")
        break
    else:
        print("Неверный выбор")
```

**Разбор:**
1. `run_schtasks` — универсальная обёртка: принимает аргументы, возвращает код и вывод
2. `/Query /TN имя` — проверяет существование задачи (код 0 = найдена)
3. `/Create /SC ONLOGON` — создаёт задачу которая запускается при входе в Windows
4. `/Delete /F` — удаляет без подтверждения
5. `while True` + `break` — цикл меню, выход только через пункт "0"

## Заметки на полях

💡 Просмотр задач в интерфейсе
> Win+R → `taskschd.msc` — откроет графический планировщик задач.
> Удобно проверять созданные задачи.

⚠️ Права администратора
> Создание задач с уровнем `HIGHEST` требует запуска от администратора.
> Для пользовательских задач используй `/RL LIMITED` — не требует прав.

📝 Альтернатива — реестр автозапуска
> Проще и без прав администратора:
> ```
> HKCU\SOFTWARE\Microsoft\Windows\CurrentVersion\Run
> ```
> Добавь строковый ключ с путём к скрипту — запустится при входе пользователя.
> Смотри урок 20 - Реестр Windows.
