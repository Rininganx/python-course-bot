# Дизайн интерфейса

> Правила отступов, цветов и иерархии для красивого UI

## Theory

Хороший UI — это отступы, иерархия и предсказуемость. Используй константы для отступов, словарь для цветов, определение шрифтов один раз. Не больше 2-3 акцентных цветов. Состояние всегда видно: загрузка, ошибка, успех.

## Code

```python
import customtkinter as ctk

# Цветовая палитра
COLORS = {
    "bg": "#1a1a2e", "surface": "#16213e",
    "accent": "#7F77DD", "success": "#a6e3a1",
    "warning": "#f9e2af", "danger": "#f38ba8",
    "text": "#cdd6f4", "subtext": "#6c7086",
}

# Шрифты (определить один раз)
FONTS = {
    "title": ctk.CTkFont(size=22, weight="bold"),
    "heading": ctk.CTkFont(size=16, weight="bold"),
    "body": ctk.CTkFont(size=13),
    "mono": ctk.CTkFont(family="Consolas", size=13),
}

# Константы отступов
PAD_SM, PAD_MD, PAD_LG = 8, 16, 24

# Карточка метрики
def make_metric_card(parent, title, value, color="#7F77DD"):
    card = ctk.CTkFrame(parent, corner_radius=12, fg_color=COLORS["surface"])
    ctk.CTkLabel(card, text=title, font=FONTS["subtext"],
                 text_color=COLORS["subtext"]).pack(padx=16, pady=(12, 0))
    val = ctk.CTkLabel(card, text=value, font=FONTS["title"], text_color=color)
    val.pack(padx=16, pady=(4, 12))
    return card, val

# Цвет по статусу
def get_status_color(value, warn=75, crit=90):
    if value >= crit: return COLORS["danger"]
    elif value >= warn: return COLORS["warning"]
    return COLORS["success"]
```

## Practice

1. Создай окно с sidebar из 4 пунктов и 3 карточками метрик (CPU, RAM, Disk)
2. Добавь кнопку "Оптимизация" и лог внизу

## Answers

```python
import customtkinter as ctk
import random

ctk.set_appearance_mode("dark")
COLORS = {"bg": "#1a1a2e", "surface": "#16213e",
          "success": "#a6e3a1", "warning": "#f9e2af", "danger": "#f38ba8"}

app = ctk.CTk()
app.geometry("600x400")
app.configure(fg_color=COLORS["bg"])

sidebar = ctk.CTkFrame(app, width=150, fg_color=COLORS["surface"], corner_radius=0)
sidebar.pack(side="left", fill="y")
sidebar.pack_propagate(False)

for item in ["Дашборд", "Оптимизация", "Профили", "Настройки"]:
    ctk.CTkButton(sidebar, text=item, fg_color="transparent",
                  anchor="w", hover_color=COLORS["surface"]).pack(fill="x", padx=5, pady=2)

main = ctk.CTkFrame(app, fg_color="transparent")
main.pack(side="right", fill="both", expand=True, padx=15, pady=15)

cards_frame = ctk.CTkFrame(main, fg_color="transparent")
cards_frame.pack(fill="x")

cards = {}
for name in ["CPU", "RAM", "Диск"]:
    card = ctk.CTkFrame(cards_frame, fg_color=COLORS["surface"], corner_radius=10)
    card.pack(side="left", padx=5, expand=True, fill="both")
    ctk.CTkLabel(card, text=name).pack(pady=(10, 0))
    val = ctk.CTkLabel(card, text="---", font=ctk.CTkFont(size=20, weight="bold"))
    val.pack()
    cards[name] = val

log_box = ctk.CTkTextbox(main, height=100)
log_box.pack(fill="x", pady=(10, 0))

def update():
    for name in ["CPU", "RAM", "Диск"]:
        v = random.randint(20, 95)
        c = COLORS["success"] if v < 60 else COLORS["warning"] if v < 80 else COLORS["danger"]
        cards[name].configure(text=f"{v}%", text_color=c)
    app.after(2000, update)

update()
app.mainloop()
```

## Tips

- Используй `fg_color="transparent"` для прозрачного фона виджета
- `sidebar.pack_propagate(False)` — фиксирует ширину sidebar
- Определяй шрифты и цвета один раз через словарь, а не в каждом виджете
