## Карточка метрики

```python
import customtkinter as ctk

def make_metric_card(parent, title: str, value: str, color: str = "#7F77DD"):
    """Создаёт карточку с метрикой."""
    card = ctk.CTkFrame(parent, corner_radius=12, fg_color="#16213e")

    ctk.CTkLabel(card, text=title,
                 font=ctk.CTkFont(size=11),
                 text_color="#6c7086").pack(padx=16, pady=(12, 0))

    val_lbl = ctk.CTkLabel(card, text=value,
                            font=ctk.CTkFont(size=28, weight="bold"),
                            text_color=color)
    val_lbl.pack(padx=16, pady=(4, 12))

    return card, val_lbl  # возвращаем val_lbl чтобы обновлять

# Использование
parent = ctk.CTkFrame(app)
parent.pack(fill="x", padx=20, pady=10)
parent.columnconfigure((0,1,2), weight=1)

card_cpu, cpu_val = make_metric_card(parent, "CPU", "72.5°C")
card_cpu.grid(row=0, column=0, padx=8, sticky="ew")

card_ram, ram_val = make_metric_card(parent, "RAM", "67%", "#a6e3a1")
card_ram.grid(row=0, column=1, padx=8, sticky="ew")

card_disk, disk_val = make_metric_card(parent, "ДИСК", "54%", "#f9e2af")
card_disk.grid(row=0, column=2, padx=8, sticky="ew")
```

## Sidebar навигация

```python
import customtkinter as ctk

class Sidebar(ctk.CTkFrame):
    def __init__(self, parent, nav_items: list, on_select):
        super().__init__(parent, width=200, corner_radius=0, fg_color="#16213e")
        self.on_select = on_select
        self.buttons   = {}
        self.active    = None

        # Логотип
        ctk.CTkLabel(self, text="⚡ PC Booster",
                     font=ctk.CTkFont(size=16, weight="bold"),
                     text_color="#7F77DD").pack(pady=24, padx=16)

        # Разделитель
        ctk.CTkFrame(self, height=1, fg_color="#2b2d42").pack(fill="x", padx=16)

        # Кнопки навигации
        for icon, name in nav_items:
            btn = ctk.CTkButton(
                self,
                text=f"  {icon}  {name}",
                anchor="w",
                fg_color="transparent",
                hover_color="#0f3460",
                corner_radius=8,
                font=ctk.CTkFont(size=13),
                command=lambda n=name: self._select(n),
            )
            btn.pack(fill="x", padx=8, pady=2)
            self.buttons[name] = btn

    def _select(self, name: str):
        # Убрать выделение с предыдущей
        if self.active:
            self.buttons[self.active].configure(fg_color="transparent")
        # Выделить новую
        self.buttons[name].configure(fg_color="#0f3460")
        self.active = name
        self.on_select(name)

# Использование
NAV_ITEMS = [
    ("📊", "Монитор"),
    ("⚡", "Оптимизация"),
    ("🚀", "Автозагрузка"),
    ("🎮", "Игровой режим"),
    ("⚙️", "Настройки"),
]

app = ctk.CTk()
app.geometry("700x500")

sidebar = Sidebar(app, NAV_ITEMS, on_select=lambda n: print(f"Открыт: {n}"))
sidebar.pack(side="left", fill="y")

content = ctk.CTkFrame(app, fg_color="transparent")
content.pack(side="left", fill="both", expand=True)

app.mainloop()
```

## Задание

PC Booster отображает карточки метрик с динамическими цветами. Создай дизайн главного экрана.

**Дано:**
- Константы `COLORS` из урока (словарь с цветами)
- Константы `FONTS` (словарь со шрифтами)
- Три карточки: CPU, RAM, Disk

**Нужно:**
1. Sidebar с 4 пунктами навигации
2. 3 карточки метрик с динамическими цветами (зелёный → жёлтый → красный)
3. Кнопка "Быстрая оптимизация"
4. Лог внизу

**Ожидаемый вид:**
```
┌─ Sidebar ─┬─────────────────────────┐
│ Дашборд   │  [CPU 45%] [RAM 67%]   │
│ Оптимизация│  [Диск 54%]            │
│ Профили   │  [Быстрая оптимизация]  │
│ Настройки │  ┌─ Лог ──────────────┐ │
│           │  │ Запуск оптимизации  │ │
└───────────┴──┴─────────────────────┘─┘
```

💡 Подсказка 1
> Используй `ctk.CTkFrame` для карточек и sidebar. Каждая карточка — отдельный фрейм с меткой и значением.

💡 Подсказка 2
> Цвет карточки зависит от значения: `fg_color = COLORS["success"] if val < 60 else COLORS["warning"] if val < 80 else COLORS["danger"]`

💡 Подсказка 3
> Обновление данных каждые 2 секунды: `app.after(2000, update_metrics)` — рекурсивный вызов.

## Решение

```python
import customtkinter as ctk
import random

ctk.set_appearance_mode("dark")
ctk.set_default_color_theme("blue")

COLORS = {
    "bg":      "#1a1a2e",
    "surface": "#16213e",
    "success": "#a6e3a1",
    "warning": "#f9e2af",
    "danger":  "#f38ba8",
    "text":    "#cdd6f4",
}

app = ctk.CTk()
app.title("PC Booster")
app.geometry("600x400")
app.configure(fg_color=COLORS["bg"])

# Sidebar
sidebar = ctk.CTkFrame(app, width=150, fg_color=COLORS["surface"], corner_radius=0)
sidebar.pack(side="left", fill="y")
sidebar.pack_propagate(False)

for item in ["Дашборд", "Оптимизация", "Профили", "Настройки"]:
    ctk.CTkButton(sidebar, text=item, fg_color="transparent",
                  anchor="w", hover_color=COLORS["surface"]).pack(fill="x", padx=5, pady=2)

# Main area
main = ctk.CTkFrame(app, fg_color="transparent")
main.pack(side="right", fill="both", expand=True, padx=15, pady=15)

# Карточки метрик
cards_frame = ctk.CTkFrame(main, fg_color="transparent")
cards_frame.pack(fill="x")

cards = {}
for name in ["CPU", "RAM", "Диск"]:
    card = ctk.CTkFrame(cards_frame, fg_color=COLORS["surface"], corner_radius=10, width=130, height=80)
    card.pack(side="left", padx=5, expand=True, fill="both")
    lbl = ctk.CTkLabel(card, text=name, font=ctk.CTkFont(size=12), text_color=COLORS["text"])
    lbl.pack(pady=(10, 0))
    val = ctk.CTkLabel(card, text="---", font=ctk.CTkFont(size=20, weight="bold"))
    val.pack()
    cards[name] = (card, val)

# Лог
log_box = ctk.CTkTextbox(main, height=100)
log_box.pack(fill="x", pady=(10, 0))

def get_color(val):
    if val < 60:
        return COLORS["success"]
    elif val < 80:
        return COLORS["warning"]
    return COLORS["danger"]

def update_metrics():
    cpu  = random.randint(20, 95)
    ram  = random.randint(40, 90)
    disk = random.randint(30, 80)

    for name, value in [("CPU", cpu), ("RAM", ram), ("Диск", disk)]:
        card, val_lbl = cards[name]
        val_lbl.configure(text=f"{value}%", text_color=get_color(value))

    app.after(2000, update_metrics)

update_metrics()
app.mainloop()
```

**Разбор:**
1. `sidebar.pack_propagate(False)` → фиксированная ширина sidebar, не растягивается
2. `fg_color="transparent"` → прозрачный фон, виден фон родителя
3. `corner_radius=10` → закруглённые углы карточек
4. `get_color(val)` → выбирает цвет по порогам: < 60 зелёный, < 80 жёлтый, иначе красный
5. `app.after(2000, update_metrics)` → рекурсивный таймер: обновляет данные каждые 2 секунды

## Заметки на полях

💡 5 правил хорошего UI
> 1. Отступы везде одинаковые — используй константы
> 2. Не больше 2-3 акцентных цветов
> 3. Иерархия: заголовок > подзаголовок > текст
> 4. Интерактивные элементы визуально отличаются от статичных
> 5. Состояние всегда видно — загрузка, ошибка, успех
