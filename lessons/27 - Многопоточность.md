## Главная идея

GUI приложение зависает если запустить долгую операцию в основном потоке. Сканирование диска, оптимизация, загрузка данных — всё это должно работать в отдельном потоке, пока интерфейс остаётся живым.

## Проблема и решение

```python
# ПЛОХО — интерфейс зависнет на 5 секунд
def optimize():
    time.sleep(5)   # долгая операция
    print("Готово")

btn = ctk.CTkButton(app, text="Оптимизировать", command=optimize)

# ХОРОШО — в отдельном потоке
import threading

def optimize():
    time.sleep(5)   # выполняется в фоне
    print("Готово")

def start_optimize():
    t = threading.Thread(target=optimize, daemon=True)
    t.start()

btn = ctk.CTkButton(app, text="Оптимизировать", command=start_optimize)
```

## threading.Thread

```python
import threading
import time

def long_task(name: str, duration: int):
    print(f"[{name}] Начало")
    time.sleep(duration)
    print(f"[{name}] Готово")

# Запустить в фоне
t = threading.Thread(target=long_task, args=("Очистка", 3), daemon=True)
t.start()

# daemon=True — поток завершится вместе с программой
# t.join()    — подождать завершения потока
```

## Обновление UI из потока

⚠️ Нельзя менять виджеты из другого потока
> tkinter и customtkinter не потокобезопасны. Изменение виджетов из фонового потока вызывает случайные крэши.
> Используй `root.after(0, func)` для передачи результата в главный поток.

```python
import customtkinter as ctk
import threading
import time

app = ctk.CTk()
app.geometry("300x200")

status_lbl = ctk.CTkLabel(app, text="Готов")
status_lbl.pack(pady=30)

progress = ctk.CTkProgressBar(app, width=250)
progress.pack()
progress.set(0)

def update_ui(text: str, value: float):
    """Безопасное обновление UI — вызывается через after()."""
    status_lbl.configure(text=text)
    progress.set(value)

def scan_task():
    steps = ["Анализ реестра...", "Сканирование temp...",
             "Проверка служб...", "Анализ автозагрузки...", "Готово!"]
    for i, step in enumerate(steps):
        app.after(0, update_ui, step, (i + 1) / len(steps))
        time.sleep(1)

def start_scan():
    t = threading.Thread(target=scan_task, daemon=True)
    t.start()

ctk.CTkButton(app, text="Сканировать", command=start_scan).pack(pady=20)
app.mainloop()
```

## Queue — безопасная передача данных

```python
import threading
import queue
import customtkinter as ctk
import psutil
import time

class MonitorThread(threading.Thread):
    """Поток мониторинга системы."""

    def __init__(self, data_queue: queue.Queue, interval: float = 1.0):
        super().__init__(daemon=True)
        self.queue    = data_queue
        self.interval = interval
        self.running  = True

    def run(self):
        while self.running:
            data = {
                "cpu":  psutil.cpu_percent(interval=None),
                "ram":  psutil.virtual_memory().percent,
                "disk": psutil.disk_usage("C:\\").percent,
            }
            self.queue.put(data)
            time.sleep(self.interval)

    def stop(self):
        self.running = False

class App(ctk.CTk):
    def __init__(self):
        super().__init__()
        self.title("Монитор")
        self.geometry("300x180")

        self.queue  = queue.Queue()
        self.thread = MonitorThread(self.queue)

        self.cpu_lbl  = ctk.CTkLabel(self, text="CPU:  ---%", font=ctk.CTkFont("Consolas", 14))
        self.ram_lbl  = ctk.CTkLabel(self, text="RAM:  ---%", font=ctk.CTkFont("Consolas", 14))
        self.disk_lbl = ctk.CTkLabel(self, text="Disk: ---%", font=ctk.CTkFont("Consolas", 14))

        for lbl in (self.cpu_lbl, self.ram_lbl, self.disk_lbl):
            lbl.pack(pady=8)

        self.thread.start()
        self._poll()
        self.protocol("WM_DELETE_WINDOW", self._on_close)

    def _poll(self):
        """Читает данные из очереди и обновляет UI."""
        try:
            while True:
                data = self.queue.get_nowait()
                self.cpu_lbl.configure(text=f"CPU:  {data['cpu']:>5.1f}%")
                self.ram_lbl.configure(text=f"RAM:  {data['ram']:>5.1f}%")
                self.disk_lbl.configure(text=f"Disk: {data['disk']:>5.1f}%")
        except queue.Empty:
            pass
        self.after(500, self._poll)

    def _on_close(self):
        self.thread.stop()
        self.destroy()

App().mainloop()
```

## Задание

PC Booster показывает прогресс оптимизации в реальном времени. Сделай кнопку с прогрессбаром.

**Дано:**
- Кнопка "Оптимизировать"
- Прогрессбар и текстовый статус

**Нужно:**
1. При нажатии кнопка блокируется (`state="disabled"`)
2. Фоновый поток делает 5 шагов по 1 секунде
3. Каждый шаг обновляет прогрессбар и текст
4. По завершении — кнопка активна, текст "Готово"

**Ожидаемый вид:**
```
[ Оптимизировать (заблокирована) ]
[████████████░░░░░░░░] 60%
Статус: Шаг 3/5 — очистка Temp...
```

💡 Подсказка 1
> `button.configure(state="disabled")` — блокирует кнопку. `"normal"` — разблокирует.

💡 Подсказка 2
> `root.after(0, callback)` — единственный безопасный способ обновить виджет из фонового потока.

💡 Подсказка 3
> `progress.set(step / 5)` — устанавливает значение прогрессбара от 0.0 до 1.0.

## Решение

```python
import customtkinter as ctk
import threading
import time

ctk.set_appearance_mode("dark")
ctk.set_default_color_theme("blue")

app = ctk.CTk()
app.title("PC Booster — Оптимизация")
app.geometry("350x200")

STEPS = 5

status_var = ctk.StringVar(value="Готово к работе")

def optimize_thread():
    for step in range(1, STEPS + 1):
        # Обновляем UI из главного потока
        app.after(0, lambda s=step: status_var.set(f"Шаг {s}/{STEPS} — работаю..."))
        app.after(0, lambda s=step: progress.set(s / STEPS))
        time.sleep(1)

    app.after(0, lambda: status_var.set("Готово!"))
    app.after(0, lambda: btn.configure(state="normal", text="Оптимизировать"))

def start_optimize():
    btn.configure(state="disabled", text="Работаю...")
    progress.set(0)
    t = threading.Thread(target=optimize_thread, daemon=True)
    t.start()

btn = ctk.CTkButton(app, text="Оптимизировать", command=start_optimize, width=250)
btn.pack(pady=(20, 10))

progress = ctk.CTkProgressBar(app, width=250)
progress.pack(pady=5)
progress.set(0)

status = ctk.CTkLabel(app, textvariable=status_var)
status.pack(pady=5)

app.mainloop()
```

**Разбор:**
1. `threading.Thread(target=..., daemon=True)` → запускает функцию в фоновом потоке
2. `daemon=True` → поток убивается вместе с приложением при закрытии
3. `app.after(0, callback)` → передаёт callback в главный поток (безопасно для tkinter)
4. `time.sleep(1)` → имитация долгой операции (sleep работает в фоновом потоке, не замораживает UI)
5. `lambda s=step:` → замыкание с захватом текущего значения `step` (без этого все лямбды увидят последнее значение)

## Заметки на полях

💡 daemon=True — всегда для фоновых потоков
> Без `daemon=True` программа не закроется пока поток не завершится.
> С `daemon=True` — поток убивается вместе с главным процессом.

⚠️ Не обновляй виджеты напрямую из потока
> Единственный безопасный способ — `root.after(0, callback)` или `Queue`.
> Прямые вызовы `.configure()` из потока приведут к крэшу — иногда не сразу, иногда через час работы.

📝 threading.Event — сигнал остановки
> ```python
> stop_event = threading.Event()
>
> def worker():
>     while not stop_event.is_set():
>         do_work()
>
> stop_event.set()   # остановить поток
> ```
