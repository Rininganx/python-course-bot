## Практический пример: Парсер цен

```python
import requests
from bs4 import BeautifulSoup
import csv

def parse_products(url):
    """Парсинг товаров с сайта."""
    headers = {"User-Agent": "Mozilla/5.0"}
    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.text, "html.parser")

    products = []
    for item in soup.select(".product-card"):
        name = item.select_one(".title").text.strip()
        price = item.select_one(".price").text.strip()
        link = item.select_one("a").get("href")
        products.append({"name": name, "price": price, "link": link})

    return products

def save_to_csv(products, filename="products.csv"):
    """Сохранение в CSV."""
    with open(filename, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=["name", "price", "link"])
        writer.writeheader()
        writer.writerows(products)

# Использование
products = parse_products("https://example.com/shop")
save_to_csv(products)
print(f"Спарсено {len(products)} товаров")
```

## Selenium для динамических страниц

```python
from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

# Запуск браузера
driver = webdriver.Chrome()
driver.get("https://example.com")

# Ждать загрузки элемента
wait = WebDriverWait(driver, 10)
element = wait.until(
    EC.presence_of_element_located((By.CLASS_NAME, "dynamic-content"))
)

# Клик по кнопке
button = driver.find_element(By.CSS_SELECTOR, ".load-more")
button.click()

# Получить данные
items = driver.find_elements(By.CSS_SELECTOR, ".item")
for item in items:
    print(item.text)

driver.quit()
```

## Задание

1. Спарси заголовки статей с любого новостного сайта
2. Сохрани результат в CSV файл
3. Добавь обработку ошибок (timeout, 404)

Используй `requests.get(url, timeout=10)` для таймаута. Проверяй `response.status_code == 200` перед парсингом.

────────────────────
## Решение

```python
import requests
from bs4 import BeautifulSoup
import csv

def parse_news(url):
    """Парсинг заголовков новостей."""
    try:
        headers = {"User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64)"}
        response = requests.get(url, headers=headers, timeout=10)
        response.raise_for_status()  # Проверка на ошибки HTTP

        soup = BeautifulSoup(response.text, "html.parser")
        articles = []

        for item in soup.select("article, .post, .news-item"):
            title_elem = item.select_one("h1, h2, h3, .title")
            if title_elem:
                title = title_elem.text.strip()
                link_elem = item.select_one("a")
                link = link_elem.get("href", "") if link_elem else ""
                articles.append({"title": title, "link": link})

        return articles

    except requests.RequestException as e:
        print(f"Ошибка запроса: {e}")
        return []

def save_news(articles, filename="news.csv"):
    """Сохранение в CSV."""
    with open(filename, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=["title", "link"])
        writer.writeheader()
        writer.writerows(articles)
    print(f"Сохранено {len(articles)} статей в {filename}")

# Использование
news = parse_news("https://news.ycombinator.com")
if news:
    save_news(news)
```
