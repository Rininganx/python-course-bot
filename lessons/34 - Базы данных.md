# Базы данных SQLite

> Структурированное хранение данных без сервера

## Theory

SQLite — встроенная БД Python, не требует сервера. Файл `.db` хранит таблицы. CRUD: Create (вставка), Read (чтение), Update (обновление), Delete (удаление). Используй параметризованные запросы `?` для защиты от SQL injection.

## Code

```python
import sqlite3

# Подключение и создание таблицы
conn = sqlite3.connect("database.db")
conn.row_factory = sqlite3.Row  # доступ по имени столбца
cursor = conn.cursor()

cursor.execute("""
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        age INTEGER
    )
""")
conn.commit()

# CRUD операции
cursor.execute("INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
    ("Андрей", "andrey@test.com", 25))
conn.commit()

cursor.execute("SELECT * FROM users WHERE age > ?", (18,))
for user in cursor.fetchall():
    print(f"{user['name']}: {user['age']}")

cursor.execute("UPDATE users SET age = ? WHERE name = ?", (26, "Андрей"))
conn.commit()

cursor.execute("DELETE FROM users WHERE name = ?", ("Андрей",))
conn.commit()
conn.close()
```

```python
# SQLAlchemy ORM
from sqlalchemy import create_engine, Column, Integer, String
from sqlalchemy.orm import declarative_base, sessionmaker

Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True)
    name = Column(String, nullable=False)
    age = Column(Integer)

engine = create_engine("sqlite:///app.db")
Base.metadata.create_all(engine)
Session = sessionmaker(bind=engine)
session = Session()

session.add(User(name="Андрей", age=25))
session.commit()
users = session.query(User).filter(User.age > 18).all()
```

## Practice

1. Создай класс `ProductsDB` с таблицей товаров (название, цена, количество)
2. Добавь CLI меню: добавить, показать, найти по цене, обновить, удалить

## Answers

```python
import sqlite3

class ProductsDB:
    def __init__(self, db="products.db"):
        self.conn = sqlite3.connect(db)
        self.conn.row_factory = sqlite3.Row
        self.conn.execute("""CREATE TABLE IF NOT EXISTS products (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name TEXT NOT NULL, price REAL, quantity INTEGER DEFAULT 0)""")
        self.conn.commit()

    def add(self, name, price, qty):
        self.conn.execute("INSERT INTO products (name, price, quantity) VALUES (?,?,?)",
            (name, price, qty))
        self.conn.commit()

    def get_all(self):
        return self.conn.execute("SELECT * FROM products").fetchall()

    def find_by_price(self, min_p, max_p):
        return self.conn.execute("SELECT * FROM products WHERE price BETWEEN ? AND ?",
            (min_p, max_p)).fetchall()

    def update_qty(self, pid, qty):
        self.conn.execute("UPDATE products SET quantity = ? WHERE id = ?", (qty, pid))
        self.conn.commit()

    def delete(self, pid):
        self.conn.execute("DELETE FROM products WHERE id = ?", (pid,))
        self.conn.commit()

db = ProductsDB()
db.add("Молоко", 80, 10)
for p in db.get_all():
    print(f"{p['name']} - {p['price']} руб. ({p['quantity']} шт.)")
```

## Tips

- Всегда `conn.commit()` после INSERT/UPDATE/DELETE
- `?` в запросах — защита от SQL injection, никогда не подставляй строки напрямую
- `row_factory = sqlite3.Row` — позволяет обращаться к столбцам по имени
