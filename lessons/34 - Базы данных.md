## Главная идея

База данных — структурированное хранилище данных. SQLite — встроенная БД Python, не требует сервера. Идеальна для ботов, десктопных приложений и прототипов.

## Подключение и создание таблиц

```python
import sqlite3

# Подключение (создаёт файл если нет)
conn = sqlite3.connect("database.db")
cursor = conn.cursor()

# Создание таблицы
cursor.execute("""
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        age INTEGER,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
""")
conn.commit()
```

## CRUD операции

```python
# Create — вставка данных
cursor.execute(
    "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
    ("Андрей", "andrey@test.com", 25)
)
conn.commit()

# Read — чтение данных
cursor.execute("SELECT * FROM users")
users = cursor.fetchall()
for user in users:
    print(user)

# Update — обновление
cursor.execute(
    "UPDATE users SET age = ? WHERE name = ?",
    (26, "Андрей")
)
conn.commit()

# Delete — удаление
cursor.execute("DELETE FROM users WHERE name = ?", ("Андрей",))
conn.commit()
```

## Контекстный менеджер

```python
import sqlite3

def get_connection():
    """Контекстный менеджер для подключения."""
    conn = sqlite3.connect("database.db")
    conn.row_factory = sqlite3.Row  # Доступ по имени столбца
    try:
        yield conn
    finally:
        conn.close()

# Использование
with get_connection() as conn:
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM users WHERE age > ?", (18,))
    adults = cursor.fetchall()
    for user in adults:
        print(f"{user['name']}: {user['age']} лет")
```

## ORM — SQLAlchemy (минимальный)

```python
from sqlalchemy import create_engine, Column, Integer, String
from sqlalchemy.orm import declarative_base, sessionmaker

Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True)
    name = Column(String, nullable=False)
    email = Column(String, unique=True)
    age = Column(Integer)

# Создание БД
engine = create_engine("sqlite:///app.db")
Base.metadata.create_all(engine)

# Сессия
Session = sessionmaker(bind=engine)
session = Session()

# Добавление
user = User(name="Андрей", email="andrey@test.com", age=25)
session.add(user)
session.commit()

# Чтение
users = session.query(User).filter(User.age > 18).all()
for u in users:
    print(f"{u.name}: {u.age}")

# Обновление
user = session.query(User).filter_by(name="Андрей").first()
user.age = 26
session.commit()

# Удаление
session.delete(user)
session.commit()
```

## Практический пример: Заметки

```python
import sqlite3
from datetime import datetime

class NotesDB:
    def __init__(self, db_path="notes.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.row_factory = sqlite3.Row
        self.create_table()

    def create_table(self):
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS notes (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                title TEXT NOT NULL,
                content TEXT,
                tags TEXT,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
                updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        self.conn.commit()

    def add_note(self, title, content, tags=""):
        self.conn.execute(
            "INSERT INTO notes (title, content, tags) VALUES (?, ?, ?)",
            (title, content, tags)
        )
        self.conn.commit()

    def get_all_notes(self):
        cursor = self.conn.execute("SELECT * FROM notes ORDER BY created_at DESC")
        return cursor.fetchall()

    def search_notes(self, query):
        cursor = self.conn.execute(
            "SELECT * FROM notes WHERE title LIKE ? OR content LIKE ? OR tags LIKE ?",
            (f"%{query}%",) * 3
        )
        return cursor.fetchall()

    def update_note(self, note_id, title=None, content=None, tags=None):
        updates = []
        params = []
        if title:
            updates.append("title = ?")
            params.append(title)
        if content:
            updates.append("content = ?")
            params.append(content)
        if tags:
            updates.append("tags = ?")
            params.append(tags)

        if updates:
            updates.append("updated_at = ?")
            params.append(datetime.now().isoformat())
            params.append(note_id)
            self.conn.execute(
                f"UPDATE notes SET {', '.join(updates)} WHERE id = ?",
                params
            )
            self.conn.commit()

    def delete_note(self, note_id):
        self.conn.execute("DELETE FROM notes WHERE id = ?", (note_id,))
        self.conn.commit()

# Использование
db = NotesDB()
db.add_note("Покупки", "Молоко, хлеб, масло", "покупки, еда")
db.add_note("Работа", "Доделать проект", "работа, проект")

notes = db.get_all_notes()
for note in notes:
    print(f"[{note['id']}] {note['title']}: {note['content']}")

results = db.search_notes("работа")
print(f"\nНайдено: {len(results)}")
```

## Задание

1. Создай класс для работы с таблицей "товаров" (название, цена, количество)
2. Добавь методы: добавить, найти по цене, обновить количество, удалить
3. Создай CLI интерфейс для работы с базой

Используй `input()` для CLI меню. Для поиска по цене — `WHERE price BETWEEN ? AND ?`.

────────────────────
## Решение

```python
import sqlite3

class ProductsDB:
    def __init__(self, db_path="products.db"):
        self.conn = sqlite3.connect(db_path)
        self.conn.row_factory = sqlite3.Row
        self.create_table()

    def create_table(self):
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS products (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                price REAL NOT NULL,
                quantity INTEGER DEFAULT 0
            )
        """)
        self.conn.commit()

    def add_product(self, name, price, quantity):
        self.conn.execute(
            "INSERT INTO products (name, price, quantity) VALUES (?, ?, ?)",
            (name, price, quantity)
        )
        self.conn.commit()
        print(f"Добавлен: {name}")

    def find_by_price(self, min_price, max_price):
        cursor = self.conn.execute(
            "SELECT * FROM products WHERE price BETWEEN ? AND ?",
            (min_price, max_price)
        )
        return cursor.fetchall()

    def update_quantity(self, product_id, new_quantity):
        self.conn.execute(
            "UPDATE products SET quantity = ? WHERE id = ?",
            (new_quantity, product_id)
        )
        self.conn.commit()
        print(f"Обновлено количество товара #{product_id}")

    def delete_product(self, product_id):
        self.conn.execute("DELETE FROM products WHERE id = ?", (product_id,))
        self.conn.commit()
        print(f"Удален товар #{product_id}")

    def get_all(self):
        cursor = self.conn.execute("SELECT * FROM products")
        return cursor.fetchall()

def main():
    db = ProductsDB()

    while True:
        print("\n=== Меню ===")
        print("1. Добавить товар")
        print("2. Показать все")
        print("3. Найти по цене")
        print("4. Обновить количество")
        print("5. Удалить товар")
        print("0. Выход")

        choice = input("\nВыбор: ")

        if choice == "1":
            name = input("Название: ")
            price = float(input("Цена: "))
            quantity = int(input("Количество: "))
            db.add_product(name, price, quantity)

        elif choice == "2":
            products = db.get_all()
            for p in products:
                print(f"[{p['id']}] {p['name']} - {p['price']} руб. ({p['quantity']} шт.)")

        elif choice == "3":
            min_p = float(input("Мин. цена: "))
            max_p = float(input("Макс. цена: "))
            results = db.find_by_price(min_p, max_p)
            for p in results:
                print(f"[{p['id']}] {p['name']} - {p['price']} руб.")

        elif choice == "4":
            pid = int(input("ID товара: "))
            qty = int(input("Новое количество: "))
            db.update_quantity(pid, qty)

        elif choice == "5":
            pid = int(input("ID товара: "))
            db.delete_product(pid)

        elif choice == "0":
            break

if __name__ == "__main__":
    main()
```
