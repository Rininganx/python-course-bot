## Работа с Excel (openpyxl)

```python
from openpyxl import Workbook, load_workbook

# Создание файла
wb = Workbook()
ws = wb.active
ws.title = "Продажи"

# Заголовки
ws["A1"] = "Товар"
ws["B1"] = "Цена"
ws["C1"] = "Количество"

# Данные
data = [
    ("Молоко", 80, 10),
    ("Хлеб", 50, 20),
    ("Масло", 150, 5),
]

for row in data:
    ws.append(row)

# Формулы
ws["D1"] = "Сумма"
for i in range(2, len(data) + 2):
    ws[f"D{i}"] = f"=B{i}*C{i}"

wb.save("sales.xlsx")

# Чтение файла
wb = load_workbook("sales.xlsx")
ws = wb.active
for row in ws.iter_rows(values_only=True):
    print(row)
```

## Работа с PDF (PyPDF2)

```python
from PyPDF2 import PdfReader, PdfWriter

# Чтение PDF
reader = PdfReader("document.pdf")
print(f"Страниц: {len(reader.pages)}")

# Извлечение текста
for i, page in enumerate(reader.pages):
    text = page.extract_text()
    print(f"Страница {i+1}: {text[:100]}...")

# Объединение PDF
writer = PdfWriter()
for pdf_file in ["file1.pdf", "file2.pdf"]:
    reader = PdfReader(pdf_file)
    for page in reader.pages:
        writer.add_page(page)

writer.save("merged.pdf")

# Извлечение страниц
reader = PdfReader("document.pdf")
writer = PdfWriter()
writer.add_page(reader.pages[0])  # Первая страница
writer.save("first_page.pdf")
```

## Отправка Email (smtplib)

```python
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

def send_email(to, subject, body, smtp_server, smtp_port, login, password):
    """Отправка email."""
    msg = MIMEMultipart()
    msg["From"] = login
    msg["To"] = to
    msg["Subject"] = subject
    msg.attach(MIMEText(body, "html"))

    with smtplib.SMTP_SSL(smtp_server, smtp_port) as server:
        server.login(login, password)
        server.send_message(msg)
    print(f"Отправлено: {to}")

# Использование
send_email(
    to="recipient@example.com",
    subject="Отчёт",
    body="<h1>Привет!</h1><p>Это автоматический отчёт.</p>",
    smtp_server="smtp.gmail.com",
    smtp_port=465,
    login="your@gmail.com",
    password="your_app_password"
)
```

## Практический пример: Автоотчёт из Excel

```python
from openpyxl import load_workbook
from datetime import datetime
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email import encoders

def generate_report(excel_path):
    """Генерация отчёта из Excel."""
    wb = load_workbook(excel_path)
    ws = wb.active

    total_sales = 0
    items_count = 0
    report_lines = []

    for row in ws.iter_rows(min_row=2, values_only=True):
        if row[0] is None:
            continue
        name, price, quantity = row[0], row[1], row[2]
        total = price * quantity
        total_sales += total
        items_count += quantity
        report_lines.append(f"<tr><td>{name}</td><td>{price}</td><td>{quantity}</td><td>{total}</td></tr>")

    html = f"""
    <h2>Отчёт за {datetime.now().strftime('%d.%m.%Y')}</h2>
    <table border="1" cellpadding="5">
        <tr><th>Товар</th><th>Цена</th><th>Кол-во</th><th>Сумма</th></tr>
        {''.join(report_lines)}
    </table>
    <p><b>Итого: {total_sales} руб. ({items_count} товаров)</b></p>
    """
    return html

def send_report(email, report_html, excel_path):
    """Отправка отчёта по email."""
    msg = MIMEMultipart()
    msg["From"] = "sender@gmail.com"
    msg["To"] = email
    msg["Subject"] = f"Отчёт {datetime.now().strftime('%d.%m.%Y')}"
    msg.attach(MIMEText(report_html, "html"))

    # Прикрепить Excel
    with open(excel_path, "rb") as f:
        part = MIMEBase("application", "octet-stream")
        part.set_payload(f.read())
        encoders.encode_base64(part)
        part.add_header("Content-Disposition", f"attachment; filename={excel_path}")
        msg.attach(part)

    with smtplib.SMTP_SSL("smtp.gmail.com", 465) as server:
        server.login("sender@gmail.com", "password")
        server.send_message(msg)

# Использование
report = generate_report("sales.xlsx")
send_report("boss@company.com", report, "sales.xlsx")
```

## Задание

1. Создай скрипт который читает CSV файл с данными сотрудников
2. Сгенерируй HTML отчёт с таблицей и статистикой
3. Сохрани отчёт в HTML файл

Используй `csv.DictReader` для чтения CSV. Для HTML — просто строки с тегами.

────────────────────
## Решение

```python
import csv
from datetime import datetime

def generate_employee_report(csv_path, output_html="report.html"):
    """Генерация HTML отчёта по сотрудникам."""
    employees = []

    with open(csv_path, "r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for row in reader:
            employees.append(row)

    if not employees:
        print("Нет данных")
        return

    # Статистика
    total = len(employees)
    departments = {}
    for emp in employees:
        dept = emp.get("department", "Неизвестный")
        departments[dept] = departments.get(dept, 0) + 1

    # Генерация HTML
    rows = ""
    for emp in employees:
        rows += f"""
        <tr>
            <td>{emp.get('name', '-')}</td>
            <td>{emp.get('position', '-')}</td>
            <td>{emp.get('department', '-')}</td>
            <td>{emp.get('salary', '-')}</td>
        </tr>"""

    dept_stats = "".join(
        f"<li>{dept}: {count} чел.</li>"
        for dept, count in departments.items()
    )

    html = f"""
    <!DOCTYPE html>
    <html>
    <head>
        <meta charset="UTF-8">
        <title>Отчёт по сотрудникам</title>
        <style>
            body {{ font-family: Arial; margin: 20px; }}
            table {{ border-collapse: collapse; width: 100%; }}
            th, td {{ border: 1px solid #ddd; padding: 8px; text-align: left; }}
            th {{ background: #4CAF50; color: white; }}
            tr:nth-child(even) {{ background: #f2f2f2; }}
            .stats {{ background: #e8f5e9; padding: 15px; border-radius: 5px; }}
        </style>
    </head>
    <body>
        <h1>Отчёт по сотрудникам</h1>
        <p>Дата: {datetime.now().strftime('%d.%m.%Y %H:%M')}</p>

        <div class="stats">
            <h3>Статистика</h3>
            <p>Всего сотрудников: <b>{total}</b></p>
            <h4>По отделам:</h4>
            <ul>{dept_stats}</ul>
        </div>

        <h2>Список сотрудников</h2>
        <table>
            <tr>
                <th>Имя</th>
                <th>Должность</th>
                <th>Отдел</th>
                <th>Зарплата</th>
            </tr>
            {rows}
        </table>
    </body>
    </html>
    """

    with open(output_html, "w", encoding="utf-8") as f:
        f.write(html)

    print(f"Отчёт сохранён: {output_html}")
    print(f"Всего сотрудников: {total}")
    print(f"Отделов: {len(departments)}")

# Использование
generate_employee_report("employees.csv")
```
