# Автоматизация рутины

> Excel, PDF, email — автоматизация офисных задач на Python

## Theory

Python автоматизирует рутину: обработка Excel (openpyxl), PDF (PyPDF2), email (smtplib). Второй по популярности тип заказов на фрилансе после ботов. Один скрипт может заменить часы ручной работы.

## Code

```python
# Excel — openpyxl
from openpyxl import Workbook, load_workbook

wb = Workbook()
ws = wb.active
ws["A1"] = "Товар"
ws["B1"] = "Цена"
ws.append(("Молоко", 80))
ws.append(("Хлеб", 50))
ws["C1"] = "Сумма"
ws["C2"] = "=A2*B2"  # формула
wb.save("sales.xlsx")

# Чтение
wb = load_workbook("sales.xlsx")
for row in wb.active.iter_rows(values_only=True):
    print(row)

# PDF — PyPDF2
from PyPDF2 import PdfReader, PdfWriter

reader = PdfReader("doc.pdf")
print(f"Страниц: {len(reader.pages)}")
text = reader.pages[0].extract_text()

# Объединение PDF
writer = PdfWriter()
for pdf in ["file1.pdf", "file2.pdf"]:
    for page in PdfReader(pdf).pages:
        writer.add_page(page)
writer.save("merged.pdf")

# Email — smtplib
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart

msg = MIMEMultipart()
msg["From"] = "you@gmail.com"
msg["To"] = "recipient@example.com"
msg["Subject"] = "Отчёт"
msg.attach(MIMEText("<h1>Отчёт</h1>", "html"))

with smtplib.SMTP_SSL("smtp.gmail.com", 465) as server:
    server.login("you@gmail.com", "app_password")
    server.send_message(msg)
```

## Practice

1. Создай скрипт чтения CSV и генерации HTML отчёта с таблицей и статистикой
2. Сохрани отчёт в HTML файл

## Answers

```python
import csv
from datetime import datetime

def generate_report(csv_path, output="report.html"):
    employees = []
    with open(csv_path, "r", encoding="utf-8") as f:
        for row in csv.DictReader(f):
            employees.append(row)

    depts = {}
    for emp in employees:
        d = emp.get("department", "Неизвестный")
        depts[d] = depts.get(d, 0) + 1

    rows = "".join(
        f"<tr><td>{e.get('name','-')}</td><td>{e.get('position','-')}</td>"
        f"<td>{e.get('department','-')}</td><td>{e.get('salary','-')}</td></tr>"
        for e in employees
    )

    html = f"""<!DOCTYPE html>
<html><head><meta charset="UTF-8"><title>Отчёт</title>
<style>
body {{ font-family: Arial; margin: 20px; }}
table {{ border-collapse: collapse; width: 100%; }}
th, td {{ border: 1px solid #ddd; padding: 8px; text-align: left; }}
th {{ background: #4CAF50; color: white; }}
</style></head>
<body>
<h1>Отчёт по сотрудникам</h1>
<p>{datetime.now().strftime('%d.%m.%Y %H:%M')}</p>
<p>Всего: <b>{len(employees)}</b></p>
<table><tr><th>Имя</th><th>Должность</th><th>Отдел</th><th>ЗП</th></tr>
{rows}</table></body></html>"""

    with open(output, "w", encoding="utf-8") as f:
        f.write(html)
    print(f"Отчёт: {output}")

generate_report("employees.csv")
```

## Tips

- Для Gmail используй "App Password" а не обычный пароль
- `openpyxl` поддерживает формулы Excel — пиши их как строки
- `PyPDF2` не может редактировать текст PDF — только читать и объединять
